<script setup>
import { ref, onMounted } from 'vue'

import EventService from '../../services/event.service'
import VenueService from '../../services/venue.service'

const parallax = new URL('@/assets/stadium.jpeg', import.meta.url).href

const menus = ref({})
const venues = ref([])
const groups = ref({})
const sortedEvents = ref({})

onMounted(async () => {
	const { data: events } = await EventService.all()
	const { data } = await VenueService.all()

	venues.value = data

	// Group venues by type
	sortedEvents.value = events.reduce((acc, cur) => {
		// menus.value = Array(calendarSchedule.value.length).fill([])
		if (!menus.value[cur.type]) menus.value[cur.type] = Array.from({ length: data.length }, () => false) // as many menus as events so we have one menu per event
		if (!groups.value[cur.type]) groups.value[cur.type] = null // default selected model
		if (!acc[cur.type]) acc[cur.type] = []
		acc[cur.type].push(cur)
		return acc
	}, {})
})

const onIntersect = {
	handler: (b, e) => {
		e[0].target.style.transition = 'opacity .3s ease'
		e[0].target.style.opacity = e[0].intersectionRatio
	},
	options: { threshold: [0, .25, .5, .75, 1] }
}
</script>

<template>
	<v-parallax :src="parallax">
		<div class="d-flex flex-column fill-height justify-center align-center">
			<div class="text-h2 font-weight-thin mb-4">BeSpectacled Events</div>
			<div class="text-h4 text-secondary">Discover our events</div>

			<v-btn
				color="primary"
				prepend-icon="fa fa-fade fa-computer-mouse"
				append-icon="fa fa-bounce fa-arrow-down"
				size="x-large"
				v-scroll-to="'.v-sheet'"
			/>
		</div>
	</v-parallax>

	<v-sheet v-for="key in Object.keys(sortedEvents)" :key="key" class="mx-auto" elevation="8" :class="key === Object.keys(sortedEvents)[0] ? 'mb-4' : ''">
		<div class="text-h4 font-weight-thin px-4 pt-4">For {{ key }}s</div>
		<v-slide-group v-model="groups[key]" selected-class="bg-primary border-lg" show-arrows v-intersect="onIntersect">
			<v-slide-group-item v-for="{ id, title, src } in sortedEvents[key]" :key="id" v-slot="{ isSelected, toggle, selectedClass }">
				<v-card color="grey-lighten-1" :class="['ma-4 d-flex', selectedClass]" height="200" width="200" @click="toggle">
					<v-img :src="src" cover class="d-flex fill-height align-center text-center">
						<v-scale-transition>
							<v-icon v-if="isSelected" color="white" size="48" icon="fa fa-times-circle">{{ title }}</v-icon>
						</v-scale-transition>
					</v-img>
				</v-card>
			</v-slide-group-item>
		</v-slide-group>

		<v-expand-transition>
			<v-sheet v-if="groups[key] != null" class="d-flex flex-column fill-height justify-center align-center text-white">
				<div class="text-h4 font-weight-thin">{{ sortedEvents[key][groups[key]].title }}</div>
				<!-- <div class="text-h6 text-secondary">${{ sortedEvents[key][groups[key]].price }}</div> -->

				<!-- TODO v-notify="[sortedVenues[key][groups[key]], ' succesfully booked!']" -->
				<v-btn color="primary" prepend-icon="fa fa-key fa-flip" @click="() => $router.push({ name: 'ticketing', query: { event: sortedEvents[key][groups[key]].id } })">
					Buy tickets for ${{ sortedEvents[key][groups[key]].price }} each
				</v-btn>

				<div class="text-overline">Venue it will be held at</div>

				<v-card-text>
					<v-chip-group column>
						<v-menu v-for="({ id, name, price, src }, i) in venues.filter(e => e.id === sortedEvents[key][groups[key]].venue)" :key="name" v-model="menus[key][i]" location="top start" origin="top start" transition="scale-transition">
							<template #activator="{ props }">
								<v-chip v-bind="props" pill link>
									<v-avatar start :image="src" />
									{{ name }}
								</v-chip>
							</template>

							<v-card width="max-content">
								<v-list bg-color="black">
									<v-list-item :title="name" :subtitle="`Min. $${price} per seat`" :prepend-avatar="src">
										<template #append>
											<v-list-item-action>
												<v-btn icon variant="text" @click="menus[key][i] = false">
													<v-icon icon="fa fa-times-circle" />
												</v-btn>
											</v-list-item-action>
										</template>
									</v-list-item>
								</v-list>

								<v-list>
									<v-list-item link prepend-icon="fa fa-circle-info" title="View Venue Details" @click="() => $router.push({ name: 'venue', params: { id } })" />
									<v-list-item link prepend-icon="fa fa-calendar-days" title="Check Available Times" @click="() => $router.push({ name: 'calendar', query: { venue: id } })" />
									<v-list-item link prepend-icon="fa fa-ticket" title="Book Venue" @click="() => $router.push({ name: 'ticketing', query: { venue: id } })" />
								</v-list>
							</v-card>
						</v-menu>
					</v-chip-group>
				</v-card-text>
			</v-sheet>
		</v-expand-transition>
	</v-sheet>
</template>

	height: calc(100vh - (48px + 16px * 2)) !important;
	margin-bottom: 16px;
}

	bottom: 16px;
}
